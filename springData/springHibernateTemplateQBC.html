HibernateTemplate QBC查询

QBC查询，即Criteria标准对象查询

@Controller
public class ControllerTest  {
	@Autowired
	private HibernateTemplate hibernateTemplate;
	@RequestMapping("/QBC")
    public String hibernateTemplateQBCExample() throws Exception{
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//等于条件：field1='1'
			criteria.add(Restrictions.eq("field1","1"));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//Like条件
			criteria.add(Restrictions.like("field1","1%"));
			hibernateTemplate.findByCriteria(criteria);
		}
		//Like条件
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//以什么开头
			criteria.add(Restrictions.like("field1","1", MatchMode.START));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//以什么结尾
			criteria.add(Restrictions.like("field1","1", MatchMode.END));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//任意位置
			criteria.add(Restrictions.like("field1","1", MatchMode.ANYWHERE));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			criteria.add(Restrictions.like("field1","1", MatchMode.EXACT));
			hibernateTemplate.findByCriteria(criteria);
		}
		
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			System.out.println("大于等于 >=");
			//大于等于>=条件
			criteria.add(Restrictions.ge("field1","1"));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			System.out.println("大于>");
			//大于>条件
			criteria.add(Restrictions.gt("field1","1"));
			hibernateTemplate.findByCriteria(criteria);
		}
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			System.out.println("小于<");
			//小于<条件
			criteria.add(Restrictions.geProperty("field1","1"));
			hibernateTemplate.findByCriteria(criteria);
		}
		
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			System.out.println("and条件");
			//and条件
			criteria.add(Restrictions.and(
						Restrictions.eq("field1","1"),Restrictions.eq("field1","1"),Restrictions.eq("field1","1")
					));
			hibernateTemplate.findByCriteria(criteria);
		}

		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//排序
			criteria.addOrder(Order.asc("field1"));
			hibernateTemplate.findByCriteria(criteria);
		}
		
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//分页
			hibernateTemplate.findByCriteria(criteria,0,1);
		}
		
		if(true) {
			DetachedCriteria criteria=DetachedCriteria.forClass(Table1.class);
			//count
			criteria.setProjection(Projections.rowCount());
			hibernateTemplate.findByCriteria(criteria);
		}
        return "index.jsp";
    }
}
