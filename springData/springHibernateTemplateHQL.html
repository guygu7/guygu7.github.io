HibernateTemplate HQL

HQL使用及绑定传参示例：

@Controller
public class ControllerTest  {
	@Autowired
	private HibernateTemplate hibernateTemplate;

    @RequestMapping("/HQL")
    public String hibernateTemplateHQLExample() throws Exception{
		Session s = hibernateTemplate.getSessionFactory().getCurrentSession();
		Query<Table1> query = s.createQuery("from Table1 where field1= :pramList ", Table1.class);
		//可以用数组绑定参数查询
		Object[] objPram = new Object[]{"01"};
		query.setParameterList("pramList", objPram);
		List<Table1> ls = query.getResultList();//执行查询
		
		//也可以用collection集合绑定参数查询
		List<String> listPram = new ArrayList<String>();
		listPram.add("1");
		query.setParameterList("pramList", listPram);
		List<Table1> ls2 = query.getResultList();//执行查询
		
		System.out.println(ls.toString()+ls2.toString());
		return "index.jsp";
	}
}
