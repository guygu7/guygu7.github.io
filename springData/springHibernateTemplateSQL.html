HibernateTemplate  执行原生SQL语句

传入的SQL不会被加工，而是原原本本被拿去执行。

@Controller
public class ControllerTest  {
        @RequestMapping("/SQL")
    public String hibernateTemplateSQLExample() throws Exception{
		Session s = hibernateTemplate.getSessionFactory().getCurrentSession();
		Query<Table1> query = s.createSQLQuery("select * from Table1 where field1_=?").addEntity(Table1.class);
		query.setParameter(1,"1");
		List<Table1> ls = query.getResultList();//执行查询
		System.out.println(ls.toString());
		return "index.jsp";
	}	
}
