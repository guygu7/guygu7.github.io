ref引用组件或元素

基于vue-cli的示例：

Example.vue

<template>
  <!-- 注册ref，之后可以在this中获取到，也可以引用组件 -->
  <div ref="ref3">
    <input type="button" value="获取refs" @click="getrefs" ref="ref1">
    <div ref="ref2"></div>
  </div>
</template>
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{msg:""}
  },
  methods:{// methods 属性中定义了当前 Vue 实例所有可用方法
     getrefs(){//通过 $refs 获取到组件中DOM元素的引用，只能获取本组件注册的ref，无法获取组件外的ref，包括父子组件
       window.console.info(this.$refs);
     }
  }
};
</script>
