VUE单向数据绑定

基于vue-cli

Example.vue

<template>
  <div>

    <!-- 插值表达式：在加载完成前页面会直接显示{{msg}}而不是其所等于的值，在加载完成后才会替换 -->
    <div>{{msg}}</div>
    <!-- v-cloak: 1.能够解决 插值表达式被直接显示的问题，在加载完成前，不会显示; 
                  2.不会覆盖原文本内容-->
    <div v-cloak>{{msg}}原文本内容</div>
    <!-- v-text:  1.能够解决 插值表达式被直接显示的问题，在加载完成前，不会显示; 
                  2.会覆盖原文本内容-->
    <div v-text="msg1">{{msg}}原文本内容被覆盖</div>
    <!-- html解析,不会解析html，直接转义显示 -->
    <div>{{html}}</div>
    <div v-text='html'></div>
    <!-- v-html:  1.能够解决 插值表达式被直接显示的问题，在加载完成前，不会显示; 
                  2.会覆盖原文本内容;
                  3.会将字符串的html解析 -->
    <div v-html='html'>原文本内容被覆盖</div>


    <!-- 属性绑定数据不能直接这么写 -->
    <input type='button' value=bindValue><br>
    <!-- v-bind: 属性绑定数据，可以写合法的js表达式-->
    <input type='button' v-bind:value="bindValue+'+拼接字符串{{msg}}'"><br>
    <!-- v-bind: 可以简写为 :属性 -->
    <input type='button' :value="bindValue+'+拼接字符串{{msg}}'"><br>
  </div>
</template>

//script
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
      msg:'测试数据',
      msg1:'v-text覆盖原文本',
      html:'<div>html解析测试</div>',
      bindValue:'绑定值',
    }
  }
};
</script>
