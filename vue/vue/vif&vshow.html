v-if&v-show 隐藏和显示页面元素

基于vue-cli的示例

Example.vue

<template>
    <div>
        <!-- v-if 将元素从页面移除，v-show 将元素隐藏display:none，而不移除 -->
        <input type="button" value="切换v-if v-show flag" @click="v_if_show_Flag=!v_if_show_Flag">
        <div v-if="v_if_show_Flag">v-if 隐藏时将元素从页面完全移除</div>
        <div v-show="v_if_show_Flag">v-show 将元素隐藏display:none，而不移除</div>
    </div>
</template>
//script
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
      v_if_show_Flag:true,
    }
  },
};
</script>
