LifecycleHooks生命周期钩子函数

基于vue-cli的示例：

Example.vue

<template>
  <div>
    <input type='button' value="修改值" @click="add" >
     生命周期 {{msg}}
  </div>
</template>
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{msg:"生命周期函数测试数据 data "}
  },
  methods:{// methods 属性中定义了当前 Vue 实例所有可用方法
    method(){window.console.info("生命周期函数测试方法 method ")},
    add(){this.msg+="+1"},
  },
  //生命周期函数
  beforeCreate(){//实例被创建之前
    window.console.info("生命周期函数 beforeCreate 被触发");
    window.console.log(this.msg);//data methods 等 都未初始化
    //method();//未初始化，调用将报错
  },
  created(){//实例被创建之后
    window.console.info("生命周期函数 created 被触发");
    window.console.info(this.msg);//data methods 等 已被初始化
    this.method();
  },
  beforeMount(){//挂载(渲染)到页面上之前
    window.console.info("生命周期函数 beforeMount 被触发");
  },
  mounted(){//挂载(渲染)到页面上之后
    window.console.info("生命周期函数 mounted 被触发");
  },
  beforeUpdate(){//运行时 data 更新之前
    window.console.info("生命周期函数 beforeUpdate 被触发");
  },
  updated(){//运行时 data 更新之后
    window.console.info("生命周期函数 updated 被触发");
  },
  //销毁
  beforeDestroy(){window.console.info("生命周期函数 beforeDestroy 被触发");},
  destroyed(){window.console.info("生命周期函数 destroyed 被触发");},
};
</script>
