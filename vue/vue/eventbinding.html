VUE事件绑定

基于vue-cli的示例

Example.vue

<template>
  <div>
    <!-- v-on: 事件绑定 -->
    <input type='button' value="按钮:事件绑定" v-on:click='show'><br>
    <!-- v-on: 可以简写为 @事件 -->
    <input type='button' value="按钮:事件绑定" @click='show'><br>
    <br>
    <!-- 定时器示例 -->
    <input type='button' value="开始" @click='intervalStart'>
    <input type='button' value="停止" @click='intervalStop'>
    <div>{{int}}</div>
  </div>
</template>
//script
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
        int:0,
        intervalID:null,//定时器ID
    }
  },
  methods:{// methods 属性中定义了当前 Vue 实例所有可用方法
    show:function(){
      alert("事件绑定测试");
    },
    //定时器示例方法
    intervalStart(){
      if(this.intervalID!=null)return;
      this.intervalID=setInterval(()=>{
        this.int++;//使用箭头函数，this 为 Vue 实例对象，普通function则this会指向该方法自身
      },1000)
    },
    intervalStop(){
      clearInterval(this.intervalID);
      this.intervalID=null;
    },
};
</script>
