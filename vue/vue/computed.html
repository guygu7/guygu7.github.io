computed计算属性

基于vue-cli的示例：

Example.vue

<template>
  <div>
    <div><!-- computed -->
      <input type="text" v-model="computed_value1">+
      <input type="text" v-model="computed_value2">
      <!-- 多次引用也只会依赖值变化时调用一次computed方法，不会多次调用 -->
      {{computed_value3}}<br>{{computed_value3}}<br>
    </div>
  </div>
</template>
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
        computed_value1:"",computed_value2:""
    }
  },
  computed:{//【计算属性】，本质是一个方法，但可以直接当做属性使用
    /*  1.引用时不可以加()调用
        2.仅在内部所使用data中的值被修改时调用，重新计算该属性值
        3.被调用后，返回结果会缓存，多次引用不会被多次调用 */
    computed_value3(){
      window.console.info("computed被调用");
      return this.computed_value1+this.computed_value2;//必须return
    }
  }
};
</script>
