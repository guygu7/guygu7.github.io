class和style绑定数据 动态更新

基于vue-cli的示例

Example.vue

<template>
    <div>
        <input type="button" value="切换class_Flag" @click="class_Flag=!class_Flag">
        <!-- class : 使用 v-bind   （style 和 class 相同）-->
        <div :class="['bold','italic','active']">class样式演示</div>
        <!-- 可以使用三元表达式 -->
        <div :class="['bold',class_Flag?'':'italic']">class样式演示1</div>
        <!-- 可以使用对象代替三元表达式 -->
        <div :class="['bold',{'italic':class_Flag}]">class样式演示2</div>
        <!-- 可以使用对象，对象属性是class类名，对象属性可以不带引号 -->
        <div :class="{'bold':true,italic:true,active:true}">class样式演示3</div>
        <!-- 可以使用data中保存的对象 -->
        <div :class="class_obj">class样式演示4</div>
        <!-- 可以使用数组，引用多个对象 -->
        <div :class="[class_obj,class_obj2]">class样式演示5</div>
        <!-- style 和 class 相同 -->
  </div>
</template>
//script
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
      class_Flag:true,
      class_obj:{'bold':true,italic:true},
      class_obj2:{active:true}
    }
  },
};
</script>
<style>
.bold {
  font-weight: bold;
}
.italic {
  font-style: italic;
}
.active {
  letter-spacing: 0.5em;
}
</style>
