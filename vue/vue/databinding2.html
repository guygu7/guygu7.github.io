VUE双向数据绑定

基于vue-cli

Example.vue

<template>
  <div>
    <!-- v-model 双向数据绑定  注意:只能运用在 表单 元素中 input select checkbox textarea ... -->
    <input type="text" v-model="v_model_msg" ><br>
    <div>{{v_model_msg}}</div><br>
  </div>
</template>
<script>
//关于 Vue 实例中的 this: 在Vue实例中，要获取 data 中的数据 或者 methods 中的方法，必须使用 this. 访问，这里的 this 表示 Vue 实例对象
export default {
  //name: "Example11",
  data(){//注意：使用 vue cli 后，这里是属于组件 data 必须这么写，和实例 data 不一样，实例可以是对象
    return{
      v_model_msg:'v-model 双向数据绑定',
    }
  },
};
</script>
